            <label for="sats">Sats:</label>

            <input type="number" id="sats" placeholder="Enter amount in Sats">

        </div>

        <div>

            <button onclick="convertSats()">Convert</button>

        </div>

        <div id="results"></div>

    </div>


    <script>

        async function getBitcoinPrice() {

            const response = await fetch('https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd,eur,gbp');

            const data = await response.json();

            return data.bitcoin;

        }


        async function convertSats() {

            const sats = document.getElementById('sats').value;

            const bitcoinPrices = await getBitcoinPrice();


            const satToBtc = sats / 100000000;


            const usd = (satToBtc * bitcoinPrices.usd).toFixed(2);

            const eur = (satToBtc * bitcoinPrices.eur).toFixed(2);

            const gbp = (satToBtc * bitcoinPrices.gbp).toFixed(2);


            document.getElementById('results').innerHTML = `

                <p>${sats} sats is approximately:</p>

                <p>${usd} USD</p>

                <p>${eur} EUR</p>

                <p>${gbp} GBP</p>

            `;

        }

    </script>
